**Tech Stack:** Next.js 14 App Router, TypeScript, Google Apps Script (GAS), CoinGecko API, next-intl, Tailwind CSS

```javascript
// ============================================================
// LIFAI Wallets & Sell Requests

/**
 * lifai_wallets シートから active なウォレット一覧を返す
 * レスポンス: { ok: true, data: { wallets: [{slot, wallet_address, label}] } }
 */
function getLifaiWallets() {
  var sheet = getSheet('lifai_wallets');
  var headers = ['slot', 'wallet_address', 'label', 'active'];
  ensureHeaders(sheet, headers);

  var data = sheet.getDataRange().getValues();
  if (data.length <= 1) {
    return { ok: true, code: 'OK', message: 'No wallets configured.', data: { wallets: [] } };
  }

  var rows = data.slice(1);
  var wallets = rows
    .filter(function(row) { return String(row[3]).toLowerCase() === 'true'; })
    .map(function(row) {
      return {
        slot: row[0],
        wallet_address: row[1],
        label: row[2]
      };
    });

  return { ok: true, code: 'OK', message: 'Wallets retrieved.', data: { wallets: wallets } };
}

/**
 * lifai_sell_requests シートに申請行を追加する
 * payload: { request_id, lifai_plan, ep_amount, ep_rate_jpy, usdt_rate_jpy,
 *            gross_usdt, fee_usdt, net_usdt, source_wallet, payout_network,
 *            payout_wallet, platform_wallet }
 */
function createLifaiSellRequest(payload) {
  var sheet = getSheet('lifai_sell_requests');
  var headers = [
    'request_id', 'created_at', 'lifai_plan', 'ep_amount',
    'ep_rate_jpy', 'usdt_rate_jpy', 'gross_usdt', 'fee_usdt', 'net_usdt',
    'source_wallet', 'payout_network', 'payout_wallet', 'platform_wallet', 'status'
  ];
  ensureHeaders(sheet, headers);

  var required = ['lifai_plan', 'ep_amount', 'source_wallet', 'payout_network', 'payout_wallet', 'platform_wallet'];
  for (var i = 0; i < required.length; i++) {
    if (!payload[required[i]]) {
      return { ok: false, code: 'MISSING_FIELDS', message: required[i] + ' is required.' };
    }
  }

  var now = new Date().toISOString();

  appendRow(sheet, headers, {
    request_id: payload.request_id,
    created_at: now,
    lifai_plan: payload.lifai_plan,
    ep_amount: payload.ep_amount,
    ep_rate_jpy: payload.ep_rate_jpy,
    usdt_rate_jpy: payload.usdt_rate_jpy,
    gross_usdt: payload.gross_usdt,
    fee_usdt: payload.fee_usdt,
    net_usdt: payload.net_usdt,
    source_wallet: payload.source_wallet,
    payout_network: payload.payout_network,
    payout_wallet: payload.payout_wallet,
    platform_wallet: payload.platform_wallet,
    status: 'pending'
  });

  return {
    ok: true,
    code: 'SELL_REQUEST_CREATED',
    message: 'Sell request created.',
    data: { request_id: payload.request_id }
  };
}
```

```typescript
// app/api/lifai/wallet/route.ts
import { NextResponse } from 'next/server'
import { gasClient } from '@/lib/gas-client'

const SIX_HOURS = 21_600_000

export async function GET() {
  const result = await gasClient.getLifaiWallets()

  if (!result.ok || !result.data) {
    return NextResponse.json({ ok: false, error: 'Failed to fetch wallets' }, { status: 502 })
  }

  const wallets = result.data.wallets
  if (wallets.length === 0) {
    return NextResponse.json({ ok: false, error: 'No wallets configured' }, { status: 503 })
  }

  const currentSlotNum = Math.floor(Date.now() / SIX_HOURS)
  const slotIndex = currentSlotNum % wallets.length
  const current = wallets[slotIndex]
  const nextRotateAt = new Date((currentSlotNum + 1) * SIX_HOURS).toISOString()

  return NextResponse.json({
    ok: true,
    wallet_address: current.wallet_address,
    label: current.label,
    next_rotate_at: nextRotateAt,
  })
}
```

```typescript
// app/api/lifai/sell-request/route.ts
import { NextRequest, NextResponse } from 'next/server'
import { gasClient } from '@/lib/gas-client'
import { LifaiPlan } from '@/types'

const EP_RATE_JPY: Record<LifaiPlan, number> = {
  starter: 0.25,
  builder: 0.2857,
  automation: 0.3333,
  core: 0.40,
  infra: 0.50,
}

const VALID_PLANS = Object.keys(EP_RATE_JPY) as LifaiPlan[]
const VALID_NETWORKS = ['TRC20', 'ERC20', 'BEP20']
const FEE_RATE = 0.085
const SIX_HOURS = 21_600_000

function makeRequestId(): string {
  return 'LIFAI-' + Math.random().toString(36).slice(2, 8).toUpperCase()
}

async function fetchJpyPerUsdt(): Promise<number> {
  const response = await fetch(
    'https://api.coingecko.com/api/v3/simple/price?ids=tether&vs_currencies=jpy',
    { cache: 'no-store' }
  )
  if (!response.ok) throw new Error('CoinGecko fetch failed')
  const data = await response.json() as { tether: { jpy: number } }
  return data.tether.jpy
}

export async function POST(req: NextRequest) {
  const body = await req.json() as {
    lifai_plan: string
    ep_amount: number
    source_wallet: string
    payout_network: string
    payout_wallet: string
  }

  const { lifai_plan, ep_amount, source_wallet, payout_network, payout_wallet } = body

  // バリデーション
  if (!VALID_PLANS.includes(lifai_plan as LifaiPlan)) {
    return NextResponse.json({ ok: false, error: 'Invalid lifai_plan' }, { status: 400 })
  }
  if (!ep_amount || ep_amount < 100) {
    return NextResponse.json({ ok: false, error: 'ep_amount must be at least 100' }, { status: 400 })
  }
  if (!source_wallet?.trim()) {
    return NextResponse.json({ ok: false, error: 'source_wallet is required' }, { status: 400 })
  }
  if (!VALID_NETWORKS.includes(payout_network)) {
    return NextResponse.json({ ok: false, error: 'Invalid payout_network' }, { status: 400 })
  }
  if (!payout_wallet?.trim()) {
    return NextResponse.json({ ok: false, error: 'payout_wallet is required' }, { status: 400 })
  }

  // 為替レート取得
  let jpyPerUsdt: number
  try {
    jpyPerUsdt = await fetchJpyPerUsdt()
  } catch {
    return NextResponse.json({ ok: false, error: 'Failed to fetch exchange rate' }, { status: 502 })
  }

  // 現在のウォレット取得
  const walletResult = await gasClient.getLifaiWallets()
  if (!walletResult.ok || !walletResult.data || walletResult.data.wallets.length === 0) {
    return NextResponse.json({ ok: false, error: 'No platform wallet available' }, { status: 503 })
  }
  const wallets = walletResult.data.wallets
  const slotIndex = Math.floor(Date.now() / SIX_HOURS) % wallets.length
  const platformWallet = wallets[slotIndex].wallet_address

  // 計算
  const plan = lifai_plan as LifaiPlan
  const epRateJpy = EP_RATE_JPY[plan]
  const grossJpy = ep_amount * epRateJpy
  const grossUsdt = grossJpy / jpyPerUsdt
  const feeUsdt = grossUsdt * FEE_RATE
  const netUsdt = grossUsdt - feeUsdt

  const requestId = makeRequestId()

  // GASに保存
  const saveResult = await gasClient.createLifaiSellRequest({
    request_id: requestId,
    lifai_plan: plan,
    ep_amount: ep_amount,
    ep_rate_jpy: epRateJpy,
    usdt_rate_jpy: jpyPerUsdt,
    gross_usdt: Number(grossUsdt.toFixed(4)),
    fee_usdt: Number(feeUsdt.toFixed(4)),
    net_usdt: Number(netUsdt.toFixed(4)),
    source_wallet: source_wallet.trim(),
    payout_network,
    payout_wallet: payout_wallet.trim(),
    platform_wallet: platformWallet,
  })

  if (!saveResult.ok) {
    return NextResponse.json({ ok: false, error: 'Failed to save request' }, { status: 500 })
  }

  return NextResponse.json({
    ok: true,
    request_id: requestId,
    net_usdt: Number(netUsdt.toFixed(4)),
    gross_usdt: Number(grossUsdt.toFixed(4)),
    fee_usdt: Number(feeUsdt.toFixed(4)),
    ep_rate_jpy: epRateJpy,
    usdt_rate_jpy: jpyPerUsdt,
    platform_wallet: platformWallet,
  })
}
```

- [ ] **Step 1: components/LifaiSellRequestForm.tsx を以下で置き換え**

```typescript
'use client'

import { FormEvent, useEffect, useMemo, useState } from 'react'
import { useTranslations } from 'next-intl'
import { Link } from '@/i18n/navigation'
import { LifaiPlan } from '@/types'

const EP_RATE_JPY: Record<LifaiPlan, number> = {
  starter: 0.25,
  builder: 0.2857,
  automation: 0.3333,
  core: 0.40,
  infra: 0.50,
}

const PLANS: LifaiPlan[] = ['starter', 'builder', 'automation', 'core', 'infra']
const PAYOUT_NETWORKS = ['TRC20', 'ERC20', 'BEP20']
const FEE_RATE = 0.085

type WalletInfo = {
  wallet_address: string
  label: string
  next_rotate_at: string
}

type SubmitResult = {
  request_id: string
  net_usdt: number
}

function formatCountdown(targetIso: string): string {
  const ms = new Date(targetIso).getTime() - Date.now()
  if (ms <= 0) return '0分'
  const h = Math.floor(ms / 3_600_000)
  const m = Math.floor((ms % 3_600_000) / 60_000)
  return h > 0 ? `${h}時間 ${m}分` : `${m}分`
}

export default function LifaiSellRequestForm() {
  const t = useTranslations('lifai')

  const [selectedPlan, setSelectedPlan] = useState<LifaiPlan>('starter')
  const [epAmount, setEpAmount] = useState('1000')
  const [sourceWallet, setSourceWallet] = useState('')
  const [payoutNetwork, setPayoutNetwork] = useState('TRC20')
  const [payoutWallet, setPayoutWallet] = useState('')

  const [walletInfo, setWalletInfo] = useState<WalletInfo | null>(null)
  const [walletLoading, setWalletLoading] = useState(true)
  const [walletFetchError, setWalletFetchError] = useState(false)

  const [jpyPerUsdt, setJpyPerUsdt] = useState<number | null>(null)
  const [rateUpdatedAt, setRateUpdatedAt] = useState<string>('')
  const [countdown, setCountdown] = useState('')

  const [submitting, setSubmitting] = useState(false)
  const [error, setError] = useState('')
  const [submitted, setSubmitted] = useState<SubmitResult | null>(null)

  // ウォレット情報取得
  useEffect(() => {
    setWalletLoading(true)
    setWalletFetchError(false)
    fetch('/api/lifai/wallet')
      .then((r) => r.json())
      .then((data) => {
        if (data.ok) {
          setWalletInfo(data)
        } else {
          setWalletFetchError(true)
        }
      })
      .catch(() => setWalletFetchError(true))
      .finally(() => setWalletLoading(false))
  }, [])

  // 為替レート取得（マウント時のみ。CoinGeckoはCORS対応済みなのでクライアントから直接取得）
  useEffect(() => {
    fetch('https://api.coingecko.com/api/v3/simple/price?ids=tether&vs_currencies=jpy')
      .then((r) => r.json())
      .then((data: { tether: { jpy: number } }) => {
        setJpyPerUsdt(data.tether.jpy)
        const now = new Date()
        setRateUpdatedAt(
          `${now.getHours().toString().padStart(2, '0')}:${now.getMinutes().toString().padStart(2, '0')}`
        )
      })
      .catch(() => {})
  }, [])

  // カウントダウンタイマー
  useEffect(() => {
    if (!walletInfo) return
    const tick = () => setCountdown(formatCountdown(walletInfo.next_rotate_at))
    tick()
    const id = setInterval(tick, 60_000)
    return () => clearInterval(id)
  }, [walletInfo])

  const epValue = Number(epAmount) || 0
  const epRateJpy = EP_RATE_JPY[selectedPlan]

  const { grossUsdt, feeUsdt, netUsdt } = useMemo(() => {
    if (!jpyPerUsdt || epValue <= 0) return { grossUsdt: 0, feeUsdt: 0, netUsdt: 0 }
    const gross = (epValue * epRateJpy) / jpyPerUsdt
    const fee = gross * FEE_RATE
    return {
      grossUsdt: Number(gross.toFixed(4)),
      feeUsdt: Number(fee.toFixed(4)),
      netUsdt: Number((gross - fee).toFixed(4)),
    }
  }, [epValue, epRateJpy, jpyPerUsdt])

  async function handleSubmit(event: FormEvent<HTMLFormElement>) {
    event.preventDefault()
    setError('')

    const formData = new FormData(event.currentTarget)
    const planConfirmed = formData.get('planConfirmed') === 'on'
    const transferConfirmed = formData.get('transferConfirmed') === 'on'
    const walletConfirmed = formData.get('walletConfirmed') === 'on'

    if (!planConfirmed || !transferConfirmed || !walletConfirmed) {
      setError(t('confirmError'))
      return
    }
    if (epValue < 100) {
      setError(t('epMinimumError'))
      return
    }
    if (!sourceWallet.trim() || !payoutWallet.trim()) {
      setError(t('walletError'))
      return
    }

    setSubmitting(true)
    try {
      const response = await fetch('/api/lifai/sell-request', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          lifai_plan: selectedPlan,
          ep_amount: epValue,
          source_wallet: sourceWallet.trim(),
          payout_network: payoutNetwork,
          payout_wallet: payoutWallet.trim(),
        }),
      })
      const data = await response.json()
      if (data.ok) {
        setSubmitted({ request_id: data.request_id, net_usdt: data.net_usdt })
      } else {
        setError(t('submitError'))
      }
    } catch {
      setError(t('submitError'))
    } finally {
      setSubmitting(false)
    }
  }

  return (
    <div className="grid gap-8 lg:grid-cols-[1.1fr_0.9fr]">
      <form onSubmit={handleSubmit} className="space-y-6 rounded-2xl border border-gray-200 bg-white p-6 shadow-sm">
        <div>
          <h1 className="text-2xl font-bold text-[#0b1929]">{t('sellTitle')}</h1>
          <p className="mt-2 text-sm text-gray-500">{t('sellSubtitle')}</p>
        </div>

        {/* ウォレット表示 */}
        <div className="rounded-xl border border-sky-200 bg-sky-50 px-4 py-3">
          <p className="text-xs font-semibold uppercase tracking-wide text-sky-700">{t('walletSectionTitle')}</p>
          {walletLoading ? (
            <p className="mt-1 text-sm text-gray-500">{t('walletLoading')}</p>
          ) : walletFetchError || !walletInfo ? (
            <p className="mt-1 text-sm text-red-600">{t('walletError')}</p>
          ) : (
            <>
              <p className="mt-1 font-mono text-base font-bold text-[#0b1929]">{walletInfo.wallet_address}</p>
              <p className="mt-1 text-xs text-gray-500">{t('walletNextRotate', { time: countdown })}</p>
            </>
          )}
        </div>

        {error ? <p className="rounded border border-red-200 bg-red-50 px-3 py-2 text-sm text-red-600">{error}</p> : null}
        {submitted ? (
          <p className="rounded border border-emerald-200 bg-emerald-50 px-3 py-2 text-sm text-emerald-700">
            {t('submitSuccess', { id: submitted.request_id })}
          </p>
        ) : null}

        {/* プラン選択 */}
        <div className="space-y-3">
          <p className="text-sm font-semibold text-gray-800">{t('planHeading')}</p>
          <div className="grid gap-3 sm:grid-cols-2 lg:grid-cols-3">
            {PLANS.map((plan) => (
              <label
                key={plan}
                className={`cursor-pointer rounded-xl border p-3 ${
                  selectedPlan === plan ? 'border-sky-500 bg-sky-50' : 'border-gray-200 bg-white'
                }`}
              >
                <input
                  type="radio"
                  name="plan"
                  value={plan}
                  checked={selectedPlan === plan}
                  onChange={() => setSelectedPlan(plan)}
                  className="sr-only"
                />
                <p className="text-sm font-bold text-gray-900">{t(`plans.${plan}.name`)}</p>
                <p className="mt-1 text-xs text-gray-500">{t(`plans.${plan}.desc`)}</p>
              </label>
            ))}
          </div>
        </div>

        {/* EP数量 / 送付元ウォレット */}
        <div className="grid gap-4 md:grid-cols-2">
          <div>
            <label htmlFor="lifai-ep-amount" className="mb-1 block text-sm font-medium text-gray-700">
              {t('epAmount')}
            </label>
            <input
              id="lifai-ep-amount"
              type="number"
              min={100}
              step={1}
              value={epAmount}
              onChange={(e) => setEpAmount(e.target.value)}
              className="w-full rounded border border-gray-300 px-3 py-2 text-sm outline-none focus:border-sky-400"
            />
          </div>
          <div>
            <label htmlFor="lifai-source-wallet" className="mb-1 block text-sm font-medium text-gray-700">
              {t('sourceWallet')}
            </label>
            <input
              id="lifai-source-wallet"
              type="text"
              value={sourceWallet}
              onChange={(e) => setSourceWallet(e.target.value)}
              placeholder={t('sourceWalletPlaceholder')}
              className="w-full rounded border border-gray-300 px-3 py-2 text-sm outline-none focus:border-sky-400"
            />
          </div>
        </div>

        {/* ネットワーク / 受取ウォレット */}
        <div className="grid gap-4 md:grid-cols-2">
          <div>
            <label htmlFor="lifai-payout-network" className="mb-1 block text-sm font-medium text-gray-700">
              {t('payoutNetwork')}
            </label>
            <select
              id="lifai-payout-network"
              value={payoutNetwork}
              onChange={(e) => setPayoutNetwork(e.target.value)}
              className="w-full rounded border border-gray-300 px-3 py-2 text-sm outline-none focus:border-sky-400"
            >
              {PAYOUT_NETWORKS.map((n) => (
                <option key={n} value={n}>{n}</option>
              ))}
            </select>
          </div>
          <div>
            <label htmlFor="lifai-payout-wallet" className="mb-1 block text-sm font-medium text-gray-700">
              {t('payoutWallet')}
            </label>
            <input
              id="lifai-payout-wallet"
              type="text"
              value={payoutWallet}
              onChange={(e) => setPayoutWallet(e.target.value)}
              placeholder={t('payoutWalletPlaceholder')}
              className="w-full rounded border border-gray-300 px-3 py-2 text-sm outline-none focus:border-sky-400"
            />
          </div>
        </div>

        {/* 同意チェック */}
        <div className="space-y-3 rounded-xl border border-gray-200 bg-gray-50 p-4 text-sm text-gray-600">
          <label className="flex items-start gap-2">
            <input name="planConfirmed" type="checkbox" className="mt-0.5" />
            <span>{t('planConfirmed')}</span>
          </label>
          <label className="flex items-start gap-2">
            <input name="transferConfirmed" type="checkbox" className="mt-0.5" />
            <span>{t('transferConfirmed')}</span>
          </label>
          <label className="flex items-start gap-2">
            <input name="walletConfirmed" type="checkbox" className="mt-0.5" />
            <span>{t('walletConfirmed')}</span>
          </label>
        </div>

        <button
          type="submit"
          disabled={submitting || walletLoading || !walletInfo}
          className="w-full rounded-lg bg-[#0b1929] px-4 py-3 text-sm font-bold text-white transition-colors hover:bg-[#0d2038] disabled:opacity-50"
        >
          {submitting ? '送信中...' : t('submitRequest')}

      {/* サイドパネル */}
      <div className="space-y-6">
        {/* 見込み精算サマリー */}
```

## 管理者向けGASシート初期設定メモ

実装後、以下のGASシートを手動で初期設定する必要がある:

**`lifai_wallets` シートの初期データ例:**
| slot | wallet_address | label | active |
|---|---|---|---|
| 0 | LIFAI-EP-AAAA-1111 | ウォレットA | TRUE |
| 1 | LIFAI-EP-BBBB-2222 | ウォレットB | TRUE |
| 2 | LIFAI-EP-CCCC-3333 | ウォレットC | TRUE |
| 3 | LIFAI-EP-DDDD-4444 | ウォレットD | TRUE |

4つ登録すれば6時間×4 = 24時間で1サイクル。ウォレットを追加するだけでサイクル長が自動調整される。
